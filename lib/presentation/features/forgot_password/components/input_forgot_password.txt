import 'package:easy_localization/easy_localization.dart';
import 'package:flutter/gestures.dart';
import 'package:flutter/material.dart';
import 'package:flutter_screenutil/flutter_screenutil.dart';
import 'package:mts/app/theme/app_theme.dart';
import 'package:mts/core/config/constants.dart';
import 'package:mts/form_bloc/forgot_password_form_bloc.dart.txt';
import 'package:mts/plugins/flutter_form_bloc/flutter_form_bloc.dart';
import 'package:mts/presentation/common/dialogs/loading_dialogue.dart';
import 'package:mts/presentation/common/dialogs/theme_snack_bar.dart';
import 'package:mts/presentation/common/widgets/button_bottom.dart';
import 'package:mts/presentation/common/widgets/my_text_field_bloc_builder.dart';
import 'package:mts/presentation/features/new_password/new_password_screen.dart';

class InputForgotPassword extends StatefulWidget {
  const InputForgotPassword({super.key});

  @override
  State<InputForgotPassword> createState() => _InputForgotPasswordState();
}

class _InputForgotPasswordState extends State<InputForgotPassword> {
  bool isChecked = false;

  @override
  Widget build(BuildContext context) {
    double screenWidth = MediaQuery.of(context).size.width;
    double screenHeight = MediaQuery.of(context).size.height;
    return BlocProvider(
      create: (context) => ForgotPasswordFormBloc(),
      child: Builder(
        builder: (context) {
          final forgotPasswordFormBloc = context.read<ForgotPasswordFormBloc>();
          return FormBlocListener<ForgotPasswordFormBloc, String, String>(
            onSubmitting: ((context, state) {
              FocusScope.of(context).unfocus();
              LoadingDialog.show(context);
            }),
            onSuccess: (context, state) {
              LoadingDialog.hide(context);

              ThemeSnackBar.showSnackBar(context, 'forgotPasswordOtpSend'.tr());
              // Navigate to new password page
              // Navigator.of(context).pushReplacement(MaterialPageRoute(
              //     builder: (BuildContext context) => NewPasswordScreen(
              //         email: forgotPasswordFormBloc.email.value)));
            },
            onFailure: (context, state) {
              LoadingDialog.hide(context);
              if (state.failureResponse != null) {
                ThemeSnackBar.showSnackBar(context, state.failureResponse!);
              }
            },
            onSubmissionFailed: (context, state) {
              LoadingDialog.hide(context);
            },
            child: Container(
              //height: 800.h,
              width: 448.w,
              margin: EdgeInsets.symmetric(
                horizontal: screenWidth / 3.5,
                vertical: 0.h,
              ),
              padding: EdgeInsets.symmetric(horizontal: 15.w, vertical: 10.h),
              decoration: BoxDecoration(
                color: Colors.white,
                borderRadius: BorderRadius.circular(8),
                boxShadow: [
                  BoxShadow(
                    offset: const Offset(8, 20),
                    blurRadius: 25,
                    color: Colors.black.withValues(alpha: 0.1),
                  ),
                  BoxShadow(
                    offset: const Offset(0, 10),
                    blurRadius: 10,
                    color: Colors.black.withValues(alpha: 0.04),
                  ),
                ],
              ),
              child: Padding(
                padding: const EdgeInsets.symmetric(vertical: 10),
                child: Column(
                  mainAxisAlignment: MainAxisAlignment.center,
                  crossAxisAlignment: CrossAxisAlignment.center,
                  children: [
                    // Logo
                    Container(
                      width: 178,
                      height: 178,
                      decoration: const BoxDecoration(
                        color: Colors.transparent,
                      ),
                      child: ClipRRect(
                        borderRadius: const BorderRadius.all(
                          Radius.circular(15),
                        ),
                        child: Image.asset(
                          'assets/images/logo.png',
                          fit: BoxFit.cover,
                        ),
                      ),
                    ),
                    SizedBox(height: 25.h),
                    // Header Reset Password
                    Text('resetPassword'.tr(), style: AppTheme.h1TextStyle()),
                    SizedBox(height: 25.h),
                    Padding(
                      padding: EdgeInsets.symmetric(horizontal: 10.w),
                      child:
                      //  Email Input Field
                      Column(
                        children: [
                          MyTextFieldBlocBuilder(
                            textFieldBloc: forgotPasswordFormBloc.email,
                            labelText: 'email'.tr(),
                            hintText: 'emailHint'.tr(),
                          ),
                          const SizedBox(height: 10),
                          ButtonBottom(
                            'send'.tr(),
                            press: () {
                              Navigator.of(context).pushReplacement(
                                MaterialPageRoute(
                                  builder:
                                      (BuildContext context) =>
                                          NewPasswordScreen(
                                            email:
                                                forgotPasswordFormBloc
                                                    .email
                                                    .value,
                                          ),
                                ),
                              );
                            },
                          ),
                          Row(
                            children: [
                              Padding(
                                padding: const EdgeInsets.only(
                                  top: 20,
                                  right: 20,
                                  bottom: 10,
                                ),
                                child: RichText(
                                  text: TextSpan(
                                    text: 'backToSignInForm'.tr(),
                                    style: const TextStyle(
                                      decoration: TextDecoration.underline,
                                      color: kPrimaryColor,
                                    ),
                                    recognizer:
                                        TapGestureRecognizer()
                                          ..onTap = () {
                                            NavigationUtils.pop(context);
                                          },
                                  ),
                                ),
                              ),
                            ],
                          ),
                        ],
                      ),
                    ),
                  ],
                ),
              ),
            ),
          );
        },
      ),
    );
  }
}
